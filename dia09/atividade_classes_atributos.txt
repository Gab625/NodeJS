class Livro {
    titulo: string = "" 
    autor: string = "" 
    ano_publicacao: number = ""
    preco: number = ""
    estoque: number = "" 

    constructor (titulo: string, autor: string, ano_publicacao: number,
    preco: number, estoque: number){
        this.titulo = titulo
        this.autor = autor
        this.ano_publicacao = ano_publicacao
        this.preco = preco
        this.estoque = estoque
    }
}

class Cliente {
    nome: string = ""
    email: string = ""
    telefone: number = ""
    endereco: string = ""

    constructor (nome: string, email: string, telefone: number, endereco: string){
        this.nome = nome
        this.email = email
        this.telefone = telefone
        this.endereco = endereco
    }
}
