function delta(aa, bb, cc){
  let d = (bb*bb)-4*(aa*cc)

  return d
}

function raizes(aa, bb, delta){
  let x1 = (-bb+Math.sqrt(delta))/(2*aa)
  let x2 = (-bb-Math.sqrt(delta))/(2*aa)

  return [x1, x2]
}

let a = parseFloat(prompt("Informe o valor de A: "))
let b = parseFloat(prompt("Informe o valor de B: "))
let c = parseFloat(prompt("Informe o valor de C: "))

let resultadoD = delta(a, b, c)
console.log(resultadoD)

let [resultadoR1, resultadoR2] = raizes (a, b, resultadoD)
console.log(resultadoR1, resultadoR2)
