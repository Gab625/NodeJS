n = prompt("Informe um número")
vetor = new Array(n)


for (i = 0; i < 5; i++) {
    vetor[i] = Number(prompt("Digite um número"));
}

maior = vetor[0]
menor = vetor[0]


for (let i = 0; i < 5; i++) {
    console.log("Valores digitados: "+vetor[i]);
    if (vetor[i]>maior){
      maior = vetor[i]
    }

    if (vetor[i]<menor){
      menor = vetor[i]
    }
}

console.log("Maior valor digitado: "+maior)
console.log("Menor valor digitado: "+menor)
