class Usuario {
    nome: string = ""
    login: string = ""
    senha: string = ""

    constructor (nome: string, login:string, senha:string) {
        this.nome = nome
        this.login = login
        this.senha = senha
    }
}

const usuarioGabriel: Usuario = new Usuario("Gabriel", "pradog", "230298")
const usuarioFabio: Usuario = new Usuario("Fabio", "fmarques", "456#")

console.log(usuarioGabriel)
console.log(usuarioFabio)
